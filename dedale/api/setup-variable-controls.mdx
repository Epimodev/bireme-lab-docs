---
title: setupVariableControls
description: Make your template variables editable from a dedicated panel
---

<ResponseField name="label" type="string" required>
  Label displayed above the input in variable panel
</ResponseField>

<ResponseField name="type" type="string" required>
  Type of variable used only to generate random values for preview mode.  
  You can use one of those value:
  - any
  - companyName
  - fullName
  - mailAddress
  - phoneNumber
  - postalAddress
  - url
  - date
  - time
</ResponseField>

<ResponseField name="defaultValue" type="string">
  Value displayed in preview if the input is empty in variable panel.
</ResponseField>

<ResponseField name="placeholder" type="string">
  Input placeholder in variable panel
</ResponseField>

<ResponseField name="description" type="string">
  Small description displayed below input
</ResponseField>
